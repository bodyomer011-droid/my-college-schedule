function doPost(e) { try { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); // استلام البيانات من الموقع var data = JSON.parse(e.postData.contents); // مسح البيانات القديمة (لتحديثها بالجديدة) sheet.clearContents(); // كتابة عناوين الأعمدة من جديد sheet.appendRow(["اليوم", "الوقت", "المادة", "النوع", "المكان", "الحالة"]); // كتابة البيانات الجديدة data.forEach(function(dayObj) { var dayName = dayObj.day; dayObj.sessions.forEach(function(session) { sheet.appendRow([ dayName, session.time || "", session.subject || "", session.type || "محاضرة", session.place || "", session.status || "لم تبدأ" ]); }); }); return ContentService.createTextOutput("Success").setMimeType(ContentService.MimeType.TEXT); } catch(error) { return ContentService.createTextOutput("Error: " + error.toString()).setMimeType(ContentService.MimeType.TEXT); } }
